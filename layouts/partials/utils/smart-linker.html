{{- /* Smart Linker Partial (Shared Module) Automatically injects contextual internal links into content at build time. Uses the link registry (data/internal_links.json) generated by the Python script. Usage: Replace {{ .Content }} with {{ partial "utils/smart-linker.html" . }} Features: - Density control: limits links based on word count - Only links first occurrence of each term - Respects excluded pages from overrides - Falls back to Hugo's Related Content if no registry data - Safe: only replaces text outside of HTML tags */ -}} {{- $content := .Content -}} {{- $page := . -}} {{- $wordCount := .WordCount -}} {{- /* Get registry and overrides */ -}} {{- $registry := .Site.Data.internal_links -}} {{- $overrides := .Site.Data.link_overrides -}} {{- /* Check if page should be excluded */ -}} {{- $excludedPages := $overrides.excluded_pages | default (slice) -}} {{- $isExcluded := false -}} {{- range $excludedPages -}} {{- if eq . $page.RelPermalink -}} {{- $isExcluded = true -}} {{- end -}} {{- end -}} {{- /* Skip if page is excluded or too short */ -}} {{- if or $isExcluded (lt $wordCount 300) -}} {{- $content | safeHTML -}} {{- else -}} {{- /* Calculate link budget based on word count */ -}} {{- $config := $registry.config | default (dict "max_outbound_per_1000_words" 4) -}} {{- $linksPerThousand := $config.max_outbound_per_1000_words | default 4 -}} {{- $maxLinks := div $wordCount 250 -}} {{- if lt $maxLinks 1 -}} {{- $maxLinks = 1 -}} {{- end -}} {{- $linkCount := 0 -}} {{- /* Get page data from registry */ -}} {{- $pageData := index $registry.pages $page.RelPermalink -}} {{- /* First: Process forced links from overrides */ -}} {{- with $overrides.forced_links -}} {{- range . -}} {{- if lt $linkCount $maxLinks -}} {{- $anchor := .anchor -}} {{- $target := .target -}} {{- $targetPage := site.GetPage $target -}} {{- if $targetPage -}} {{- /* Escape special regex characters */ -}} {{- $escapedAnchor := replaceRE `([.*+?^${}()|[\]\\])` `\\$1` $anchor -}} {{- /* Pattern: Match term only when it appears after > (outside tags) */ -}} {{- $pattern := printf `(>[^<]*?)(\b%s\b)([^<]*?<)` $escapedAnchor -}} {{- if findRE $pattern $content -}} {{- $link := printf `$1$2$3` $targetPage.RelPermalink -}} {{- $content = replaceRE $pattern $link $content 1 -}} {{- $linkCount = add $linkCount 1 -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Second: Process outbound targets from registry */ -}} {{- with $pageData -}} {{- range .outbound_targets -}} {{- if lt $linkCount $maxLinks -}} {{- $targetPath := .path -}} {{- $targetPage := site.GetPage $targetPath -}} {{- if $targetPage -}} {{- /* Get target's anchor terms from registry */ -}} {{- $targetData := index $registry.pages $targetPath -}} {{- if $targetData -}} {{- $linked := false -}} {{- range $targetData.anchor_terms -}} {{- if and (not $linked) (lt $linkCount $maxLinks) -}} {{- $term := . -}} {{- /* Skip very short terms and terms with special chars */ -}} {{- if and (gt (len $term) 8) (not (findRE `[<>"]` $term)) -}} {{- /* Escape special regex characters */ -}} {{- $escapedTerm := replaceRE `([.*+?^${}()|[\]\\])` `\\$1` $term -}} {{- /* Pattern: Match term only when between > and < (outside HTML tags) */ -}} {{- $pattern := printf `(>[^<]*?)\b(%s)\b` $escapedTerm -}} {{- /* Check if term exists in content */ -}} {{- if findRE $pattern $content -}} {{- /* Create link - only replace first occurrence */ -}} {{- $link := printf `$1$2` $targetPage.RelPermalink -}} {{- $content = replaceRE $pattern $link $content 1 -}} {{- $linkCount = add $linkCount 1 -}} {{- $linked = true -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Fallback: Use Hugo's Related Content if no registry data or no links made */ -}} {{- if and (eq $linkCount 0) (gt $maxLinks 0) -}} {{- $related := .Site.RegularPages.Related . | first 4 -}} {{- range $related -}} {{- if lt $linkCount $maxLinks -}} {{- $title := .Title -}} {{- /* Skip titles with special characters that could break regex */ -}} {{- if and (gt (len $title) 15) (not (findRE `[<>"]` $title)) -}} {{- $escapedTitle := replaceRE `([.*+?^${}()|[\]\\])` `\\$1` $title -}} {{- $pattern := printf `(>[^<]*?)\b(%s)\b` $escapedTitle -}} {{- if findRE $pattern $content -}} {{- $link := printf `$1$2` .RelPermalink -}} {{- $content = replaceRE $pattern $link $content 1 -}} {{- $linkCount = add $linkCount 1 -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $content | safeHTML -}} {{- end -}}
